def composite_trapezoidal(f, a, b, n):
    h = (b - a) / n
    s = f(a) + f(b)
    for i in range(1, n):
        s += 2 * f(a + i * h)
    return s * h / 2

def composite_simpsons(f, a, b, n):
    h = (b - a) / n
    s = f(a) + f(b)
    for i in range(1, n):
        if i % 2 == 0:
            s += 2 * f(a + i * h)
        else:
            s += 4 * f(a + i * h)
    return s * h / 3

def richardson_extrapolation_simp(f, a, b, n):
    I1 = composite_simpsons(f, a, b, n)
    I2 = composite_simpsons(f, a, b, 2*n)
    return (16*I2 - I1) / 15
